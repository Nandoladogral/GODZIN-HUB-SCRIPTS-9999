local player = game.Players.LocalPlayer
local teamColor = player.TeamColor
local espEnabled = false

local function isEnemy(player)
    return player.TeamColor ~= teamColor
end

local function createESP(player)
    local esp = Instance.new("TextLabel")
    esp.Name = "ESP"
    esp.Text = player.Name
    esp.Size = UDim2.new(0, 100, 0, 20)
    esp.Position = UDim2.new(0, 0, 0, 0)
    esp.BackgroundTransparency = 0.5
    esp.BackgroundColor3 = isEnemy(player) and Color3.new(1, 0, 0) or Color3.new(0, 0, 1)
    esp.Parent = game.CoreGui
end

local function updateESP()
    for _, player in pairs(game.Players:GetPlayers()) do
        if player ~= player and player.Character then
            local esp = player.Character:FindFirstChild("ESP")
            if esp then
                esp.Visible = isEnemy(player) and espEnabled
            else
                createESP(player)
            end
        end
    end
end

local function toggleESP()
    espEnabled = not espEnabled
    updateESP()
end

-- Ativar/desativar o ESP com um bot√£o
local button = Instance.new("TextButton")
button.Name = "ToggleESPButton"
button.Text = "Ativar ESP"
button.Size = UDim2.new(0, 100, 0, 30)
button.Position = UDim2.new(0, 10, 0, 10)
button.Parent = game.CoreGui

button.MouseButton1Click:Connect(toggleESP)
